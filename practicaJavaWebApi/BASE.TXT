CREATE TABLE bodega (
    ID_BOD INT PRIMARY KEY,
    PAIS VARCHAR(50)
);


CREATE TABLE repuesto (
    NOM_REP VARCHAR(30) PRIMARY KEY,
    CAN_REP INT,
    PRE_REP DOUBLE,
    ID_BOD INT,
    FOREIGN KEY (ID_BOD) REFERENCES bodega(ID_BOD)
);


SELECT R.NOM_REP, R.CAN_REP, R.PRE_REP, B.PAIS
FROM repuesto AS R
JOIN 
bodega AS B ON R.ID_BOD = B.ID_BOD
WHERE
R.NOM_REP LIKE '%PANTA%';
